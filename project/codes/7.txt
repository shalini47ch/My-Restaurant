public class Main7Activity extends AppCompatActivity {
    TextView t1, t2,t3;
    Button b1,b2;
    Button button;
    int x=0,y=0,bill=0;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main7);
        t1 = findViewById(R.id.textView22);
        t2 = findViewById(R.id.textView23);
        t3=findViewById(R.id.textView14);
        b1 = findViewById(R.id.button20);
        button = findViewById(R.id.button);
        b2=findViewById(R.id.button11);
        bill=getIntent().getIntExtra("bill3",0);
        b1.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View view) {
                Toast.makeText(Main7Activity.this,"coke is added to cart",Toast.LENGTH_SHORT).show();
                x=15;
            }
        });
        b2.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View view) {
                Toast.makeText(Main7Activity.this,"Crusher is added to cart",Toast.LENGTH_SHORT).show();
                y=60;
            }
        });
        button.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View view) {
                Toast.makeText(Main7Activity.this, "Generate bill", Toast.LENGTH_SHORT).show();
                Intent intent = new Intent(Main7Activity.this, Main8Activity.class);
                int bills=x+y+bill;
                intent.putExtra("bill4",bills);
                startActivity(intent);
            }
        });


    }
}
