public class signin1 extends AppCompatActivity {
    TextView tv1,tv2,tv3,tv4;
    EditText et1,et2,et3;
    Button b1,b2,b3;
    SQLiteDatabase db;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_signin1);
        tv1=findViewById(R.id.textView13);
        tv2=findViewById(R.id.textView17);
        tv3=findViewById(R.id.textView18);
        tv4=findViewById(R.id.textView19);
        et1=findViewById(R.id.editText);
        et2=findViewById(R.id.editText4);
        et3=findViewById(R.id.editText5);
        b1=findViewById(R.id.button14);
        b2=findViewById(R.id.button15);
        b1.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View view) {
                db=openOrCreateDatabase("customer",MODE_PRIVATE,null);
                db.execSQL("create table if not exists customer(name text,phone text(10),password text(16))");
                String nm=et1.getText().toString();
                String no=et2.getText().toString();
                String pass=et3.getText().toString();
                db.execSQL("insert into customer values('"+nm+"','"+no+"','"+pass+"')");
                Toast.makeText(signin1.this, "Move to login", Toast.LENGTH_SHORT).show();
                Intent intent=new Intent(signin1.this,Main2Activity.class);
                startActivity(intent);
            }
        });
        b2.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View view) {

                Toast.makeText(signin1.this, "Successfully signed in", Toast.LENGTH_SHORT).show();
            }
        });
        


    }
}
