public class Main5Activity extends AppCompatActivity {
    TextView tv1,tv2,tv3;
    Button b1,b2,b3,b4;
    int x=0,y=0,z=0,bill=0;
    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main5);
        tv1=findViewById(R.id.textView9);
        tv2=findViewById(R.id.textView10);
        tv3=findViewById(R.id.textView12);
        b1=findViewById(R.id.button8);
        b2=findViewById(R.id.button9);
        b3=findViewById(R.id.button10);
        b4=findViewById(R.id.button17);
        bill=getIntent().getIntExtra("billl",0);

        b1.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View view) {
                Toast.makeText(Main5Activity.this, "rolls added to cart", Toast.LENGTH_SHORT).show();
                x=40;
            }
        });
        b2.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View view) {
                Toast.makeText(Main5Activity.this, "Burger added to cart", Toast.LENGTH_SHORT).show();
                y=25;
            }
        });
        b3.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View view) {
                Toast.makeText(Main5Activity.this, "Cream roll added", Toast.LENGTH_SHORT).show();
                z=20;
            }
        });
        b4.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View view) {

                Toast.makeText(Main5Activity.this, "move to move course", Toast.LENGTH_SHORT).show();
                Intent intent=new Intent(Main5Activity.this,Main6Activity.class);
                int bi=x+y+z+bill;
                intent.putExtra("bill2",bi);
                startActivity(intent);
            }
        });
    }
}
