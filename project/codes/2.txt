public class Main2Activity extends AppCompatActivity {
    EditText username,password;
    Button log,signin;
    TextView tv1;
    SQLiteDatabase db;
    int f=0;


    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main2);
        username = findViewById(R.id.editText2);
        password =findViewById(R.id.editText3);
        log = findViewById(R.id.button);
        signin =findViewById(R.id.button2);
        tv1 = findViewById(R.id.textView2);
        db=openOrCreateDatabase("customer",MODE_PRIVATE,null);
        log.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View view) {
                String query = "select * from customer";
                Cursor cursor = db.rawQuery(query, null);
                while (cursor.moveToNext()) {
                    String nm = cursor.getString(0);
                    String ps = cursor.getString(2);
                    if (username.getText().toString().equals(nm) && password.getText().toString().equals(ps)) {
                        Toast.makeText(Main2Activity.this, "user and password is correct", Toast.LENGTH_SHORT).show();
                        Intent intent = new Intent(Main2Activity.this, Main4Activity.class);
                        startActivity(intent);
                        f=1;
                    }
                    }
                    if(f==0)
                    {

                        Toast.makeText(Main2Activity.this, "user and password not correct", Toast.LENGTH_SHORT).show();
                    }
                }

        });
        signin.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View view) {

                Toast.makeText(Main2Activity.this, "move to signin page", Toast.LENGTH_SHORT).show();
                Intent intent=new Intent(Main2Activity.this,signin1.class);
                startActivity(intent);
            }
        });


    }
}
